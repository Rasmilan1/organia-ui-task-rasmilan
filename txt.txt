export default function ShipmentTabs() {
  const [tabs, setTabs] = useState([{ id: 1 }, { id: 2 }, { id: 3 }]);
  const [activeTab, setActiveTab] = useState(1);

  const addTab = () => {
    const newId = tabs.length > 0 ? Math.max(...tabs.map((t) => t.id)) + 1 : 1;
    setTabs([...tabs, { id: newId }]);
    setActiveTab(newId);
  };

  return (
    <div className="w-full bg-[#E4EAF6] p-4">
      <div className="flex items-end -mb-0.75">
        {tabs.map((tab) => (
          <Tab
            height={12}
            key={tab.id}
            text={`T ${String(tab.id).padStart(2, "0")}`}
            isActive={activeTab === tab.id}
            onClick={() => setActiveTab(tab.id)}
            activeColor="#001B3D"
            // className="min-w-73.5 h-13"
          />
        ))}

        <button
          onClick={addTab}
          className="ml-auto mb-2 px-6 py-2 rounded-full bg-linear-to-r from-[#8B1D3B] to-[#1A2E4B] text-white text-xs font-bold"
        >
          + Add T
        </button>
      </div>

      {/* Content Container */}
      <div className="relative z-0 border-[3px] border-[#001B3D] rounded-2xl p-6 bg-white shadow-lg">
        <ShipmentForm tabId={activeTab} />
      </div>
    </div>
  );
}
